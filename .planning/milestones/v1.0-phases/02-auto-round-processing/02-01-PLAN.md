    ```typescript
    type ConditionWithOptions = {
      condition: Condition
      duration?: number
      protected: boolean
    }

    function toggleConditionWithOptions(
      creatureId: string,
      conditionWithOptions: ConditionWithOptions
    ): void {
      const creature = creatures.value.find(c => c.id === creatureId)
      if (!creature) return

      const { condition: conditionName, duration, protected: isProtected } = conditionWithOptions
      const index = creature.conditions.indexOf(conditionName)

      if (index > -1) {
        // Removing condition - also clear duration and protected tracking if tracked
        creature.conditions.splice(index, 1)
        if (creature.conditionDurations) {
          delete creature.conditionDurations[conditionName]
        }
        // Clear from global duration map
        if (creatureConditionDurations.value[creatureId]) {
          delete creatureConditionDurations.value[creatureId][conditionName]
          if (Object.keys(creatureConditionDurations.value[creatureId]).length === 0) {
            delete creatureConditionDurations.value[creatureId]
          }
        }
        // Clear protected tracking if it was protected
        if (protectedConditions.value[creatureId]?.includes(conditionName)) {
          protectedConditions.value[creatureId] = protectedConditions.value[creatureId]!.filter(
            c => c !== conditionName
          )
          if (protectedConditions.value[creatureId].length === 0) {
            delete protectedConditions.value[creatureId]
          }
        }
      } else {
        // Adding condition
        creature.conditions.push(conditionName)

        // Set duration if provided
        if (duration !== undefined && duration > 0) {
          if (!creature.conditionDurations) {
            creature.conditionDurations = {}
          }
          creature.conditionDurations[conditionName] = duration

          // Also track in global map
          if (!creatureConditionDurations.value[creatureId]) {
            creatureConditionDurations.value[creatureId] = {}
          }
          creatureConditionDurations.value[creatureId][conditionName] = duration
        }

        // Track protected flag per-instance
        if (isProtected) {
          if (!protectedConditions.value[creatureId]) {
            protectedConditions.value[creatureId] = []
          }
          protectedConditions.value[creatureId].push(conditionName)
        }
      }
    }
    ```

    Note: `protected` flag is tracked per-instance in protectedConditions map.
    Protected conditions will skip duration decrement in decrementDurationsForCreature.
