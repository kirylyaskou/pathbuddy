/**

 * These are distinct from conditionDurations (auto-decrement timers).

export const CONDITIONS_WITH_VALUES = new Set<Condition>([
  'stunned',
  'slowed',
])

export function conditionHasValue(condition: Condition): boolean {
  return CONDITIONS_WITH_VALUES.has(condition)
}

/**
 * Static badge class map — avoids Tailwind JIT dynamic class scanning failure.

  stunned:       'bg-crimson text-stone-100',

**3. Add `setConditionValue` and `getConditionValue` to `src/stores/combat.ts`:**

Add these two functions inside the `defineStore` callback, BEFORE the `return` statement (before line 309):

```typescript
function setConditionValue(id: string, condition: Condition, value: number): void {
  const creature = creatures.value.find(c => c.id === id)
  if (!creature) return
  if (value <= 0) {
    // Remove condition entirely when value reaches 0
    const index = creature.conditions.indexOf(condition)
    if (index > -1) creature.conditions.splice(index, 1)
    if (creature.conditionValues) {
      delete creature.conditionValues[condition]
    }
  } else {
    if (!creature.conditionValues) creature.conditionValues = {}
    creature.conditionValues[condition] = value
    // Ensure condition is active in the conditions array
    if (!creature.conditions.includes(condition)) {
      creature.conditions.push(condition)
    }
  }
}

function getConditionValue(id: string, condition: Condition): number {
  const creature = creatures.value.find(c => c.id === id)
  return creature?.conditionValues?.[condition] ?? 1
}
```

```vue
<script setup lang="ts">
import { ref } from 'vue'

const props = defineProps<{
  currentHp: number
  maxHp: number
  creatureId: string
}>()

const emit = defineEmits<{
  modifyHp: [delta: number]
}>()

const amount = ref(1)

const handleDamage = () => {
  emit('modifyHp', -amount.value)
}

  emit('modifyHp', amount.value)
}

```

Template structure:
```
<div class="flex flex-wrap gap-1 items-center">
  <!-- Active condition badges -->
  <template v-for="condition in conditions" :key="condition">
    <button
      @click="handleBadgeClick(condition)"
      :class="['px-2 py-0.5 rounded-full text-xs font-normal cursor-pointer', CONDITION_BADGE_CLASSES[condition]]"
      :title="formatCondition(condition)"
    >
      {{ formatCondition(condition) }}
      <span v-if="conditionHasValue(condition)">
        {{ conditionValues?.[condition] ?? 1 }}
      </span>
    </button>
  </template>

  <!-- Add condition button -->
  <div class="relative">
    <button
      @click="showPicker = !showPicker"
      class="w-6 h-6 rounded-full bg-charcoal-500 hover:bg-charcoal-400 text-stone-400 hover:text-stone-200 text-xs flex items-center justify-center transition-colors"
      aria-label="Add condition"
    >
      +
    </button>

    <!-- Condition picker popover -->
    <div
      v-if="showPicker"
      ref="pickerRef"
      class="absolute left-0 top-8 bg-charcoal-600 border border-charcoal-500 rounded-lg p-2 shadow-card z-10 grid grid-cols-2 gap-1 w-64"
    >
      <button
        v-for="def in CONDITION_DEFS"
        :key="def.name"
        @click="handlePickerSelect(def.name)"
        :class="[
          'px-2 py-1 rounded text-xs text-left truncate transition-colors',
          conditions.includes(def.name)
            ? CONDITION_BADGE_CLASSES[def.name] + ' opacity-60'
            : 'bg-charcoal-500 text-stone-400 hover:bg-charcoal-400 hover:text-stone-200'
        ]"
      >
        {{ formatCondition(def.name) }}
      </button>
    </div>
  </div>
</div>
```

Script logic:
- `import { ref, onMounted, onUnmounted } from 'vue'`
- `import { CONDITION_DEFS, formatCondition, conditionHasValue, CONDITION_BADGE_CLASSES } from '@/composables/useConditions'`
- `import type { Condition } from '@/types/combat'`
- `const showPicker = ref(false)`
- `const pickerRef = ref<HTMLElement | null>(null)`
- Click-outside handler: `document.addEventListener('mousedown', handleClickOutside)` on mounted, remove on unmounted. `handleClickOutside` checks if click target is outside `pickerRef` and the add button, closes picker if so.
- `handleBadgeClick(condition)`: For value-bearing conditions, emit `setConditionValue` with `currentValue - 1` (decrement on click; if result <= 0, the store removes it). For binary conditions, emit `toggleCondition`.
- `handlePickerSelect(condition)`: If already active, emit `toggleCondition` (remove it). If not active, emit `toggleCondition` (add it) — for value-bearing conditions, also emit `setConditionValue` with value `1`. Close picker after selection.

Important: The ConditionBadge component does NOT import from the store directly — it receives data via props and communicates via emits. The parent (CreatureCard in Plan 03) wires store calls.
