    **Step 1: Create src/components/CreatureBrowser.vue:**

    **Script setup:**
    ```typescript
    import { ref, onMounted } from 'vue'
    import { VList } from 'virtua/vue'
    import EntityFilterBar from '@/components/EntityFilterBar.vue'
    import { filterEntities } from '@/lib/entity-query'
    import type { EntityFilter, EntityResult } from '@/lib/entity-query'
    import { useCreatureDetailStore } from '@/stores/creatureDetail'

    const props = defineProps<{
      defaultEntityType?: string
    }>()

    const emit = defineEmits<{
      'row-click': [result: EntityResult]
    }>()

    const detailStore = useCreatureDetailStore()

    const results = ref<EntityResult[]>([])
    const totalCount = ref(0)
    const loading = ref(false)
    const error = ref(false)
    const lastFilter = ref<EntityFilter>({})

    let debounceTimer: ReturnType<typeof setTimeout> | null = null

    function onFilterChange(filter: EntityFilter) {
      lastFilter.value = filter
      if (debounceTimer) clearTimeout(debounceTimer)
      debounceTimer = setTimeout(() => runQuery(filter), 200)
    }

    async function runQuery(filter: EntityFilter) {
      loading.value = true
      error.value = false
      try {
        // Fetch 101 to detect if there are more than 100
        const rows = await filterEntities(filter, 101)
        if (rows.length > 100) {
          totalCount.value = rows.length  // indicates "more than 100"
          results.value = rows.slice(0, 100)
        } else {
          totalCount.value = rows.length
          results.value = rows
        }
      } catch {
        error.value = true
        results.value = []
        totalCount.value = 0
      } finally {
        loading.value = false
      }
    }

    function handleRowClick(result: EntityResult) {
      emit('row-click', result)
      // Open the existing slide-over panel with the creature's raw data
      const rawData = JSON.parse(result.rawData)
      detailStore.openCreature(rawData, result.name)
    }

    // Initial load on mount
    onMounted(() => {
      const initialFilter: EntityFilter = {}
      if (props.defaultEntityType) {
        initialFilter.entityType = props.defaultEntityType
      }
      runQuery(initialFilter)
    })
    ```
