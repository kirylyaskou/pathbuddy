```sql
  content_rowid='id',
  tokenize='unicode61'
)
```

**8. Create 3 FTS5 sync triggers:**

INSERT trigger:
```sql
CREATE TRIGGER pf2e_fts_ai AFTER INSERT ON pf2e_entities BEGIN
  INSERT INTO pf2e_fts(rowid, name, entity_type, pack)
  VALUES (new.id, new.name, new.entity_type, new.pack);
END
```

DELETE trigger:
```sql
CREATE TRIGGER pf2e_fts_ad AFTER DELETE ON pf2e_entities BEGIN
  INSERT INTO pf2e_fts(pf2e_fts, rowid, name, entity_type, pack)
  VALUES ('delete', old.id, old.name, old.entity_type, old.pack);
END
```

UPDATE trigger (CRITICAL: delete old values BEFORE inserting new):
```sql
CREATE TRIGGER pf2e_fts_au AFTER UPDATE ON pf2e_entities BEGIN
  INSERT INTO pf2e_fts(pf2e_fts, rowid, name, entity_type, pack)
  VALUES ('delete', old.id, old.name, old.entity_type, old.pack);
  INSERT INTO pf2e_fts(rowid, name, entity_type, pack)
  VALUES (new.id, new.name, new.entity_type, new.pack);
END
```

**9. Rebuild FTS5 index for pre-existing data (CRITICAL: without this, existing entities are invisible to MATCH queries):**
```sql
INSERT INTO pf2e_fts(pf2e_fts) VALUES ('rebuild')
```

Each SQL statement is a separate string in the `sql` array (matching the v1 pattern where each statement is executed individually by `sqlite.execute(sql, [])`).

Do NOT wrap the migration in a transaction — the migration runner executes each statement individually, and DDL statements in SQLite auto-commit.
