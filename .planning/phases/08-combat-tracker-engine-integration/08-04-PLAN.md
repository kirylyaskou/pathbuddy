    // Turn advancement logic — manages initiative order cycling, round tracking,
    // condition auto-decrement via engine, and snapshot for previous turn reversal.

    import { useCombatantStore } from '@/entities/combatant'
    import { useConditionStore } from '@/entities/condition'
    import { useCombatTrackerStore } from '../model/store'
    import { endTurnConditions, getManagerState, hydrateManager } from './condition-bridge'
    import type { ActiveCondition } from '@/entities/condition'
    import { toast } from 'sonner'

    // Snapshot for previous turn reversal (D-14: one step back only)
    interface TurnSnapshot {
      activeCombatantId: string | null
      round: number
      turn: number
      // Condition state before auto-decrement (for the combatant whose turn ended)
      conditionsBefore: ActiveCondition[]
      combatantId: string // which combatant's turn ended
    }

    let lastSnapshot: TurnSnapshot | null = null

    /** Advance to the next combatant's turn. Auto-decrements conditions for the ending combatant. */
    export function advanceTurn(): void {
      const combatants = useCombatantStore.getState().combatants
      const tracker = useCombatTrackerStore.getState()

      if (!tracker.isRunning || combatants.length === 0) return

      const currentIdx = combatants.findIndex((c) => c.id === tracker.activeCombatantId)
      const endingCombatantId = tracker.activeCombatantId

      // Save snapshot for previous turn reversal
      if (endingCombatantId) {
        lastSnapshot = {
          activeCombatantId: tracker.activeCombatantId,
          round: tracker.round,
          turn: tracker.turn,
          conditionsBefore: useConditionStore
            .getState()
            .activeConditions.filter((c) => c.combatantId === endingCombatantId),
          combatantId: endingCombatantId,
        }

        // Auto-decrement conditions for the ending combatant (CMB-04)
        const changes = endTurnConditions(endingCombatantId)
        if (changes.length > 0) {
          const summary = changes
            .map((c) => {
              const name = c.slug.replace('-', ' ')
              if (c.to === null) return `${name} removed`
              return `${name} ${c.from} → ${c.to}`
            })
            .join(', ')
          const combatant = combatants.find((cb) => cb.id === endingCombatantId)
          toast(`${combatant?.displayName ?? 'Combatant'}: ${summary}`)
        }
      }

      // Calculate next combatant
      let nextIdx: number
      let newRound = tracker.round
      let newTurn = tracker.turn + 1

      if (currentIdx === -1 || currentIdx >= combatants.length - 1) {
        // Wrap around to first combatant — new round
        nextIdx = 0
        newRound = tracker.round + 1
        newTurn = 0
      } else {
        nextIdx = currentIdx + 1
      }

      const nextCombatant = combatants[nextIdx]
      tracker.setActiveCombatant(nextCombatant.id)
      tracker.setRound(newRound)
      tracker.setTurn(newTurn)
    }

    /** Reverse to the previous combatant's turn. Restores condition snapshot (D-14). */
    export function reverseTurn(): void {
      if (!lastSnapshot) return

      const tracker = useCombatTrackerStore.getState()
      if (!tracker.isRunning) return

      // Restore tracker state
      tracker.setActiveCombatant(lastSnapshot.activeCombatantId)
      tracker.setRound(lastSnapshot.round)
      tracker.setTurn(lastSnapshot.turn)

      // Restore condition state for the combatant whose turn was reversed
      const { combatantId, conditionsBefore } = lastSnapshot

      // Re-hydrate ConditionManager from the snapshot
      // First clear the current state, then restore from before
      const conditionsForHydrate = conditionsBefore.map((c) => ({
        slug: c.slug as import('@engine').ConditionSlug,
        value: c.value ?? 1,
        isLocked: !!c.isLocked,
        grantedBy: c.grantedBy as import('@engine').ConditionSlug | undefined,
      }))

      // Clear and re-create the manager for this combatant
      const { clearCombatantManager } = require('./condition-bridge')
      clearCombatantManager(combatantId)
      hydrateManager(combatantId, conditionsForHydrate)

      toast('Reversed to previous turn')

      // Clear snapshot — only one step back
      lastSnapshot = null
    }

    /** Check if previous turn reversal is available. */
    export function canReverseTurn(): boolean {
      return lastSnapshot !== null
    }

    /** Clear the turn snapshot (combat end). */
    export function clearTurnSnapshot(): void {
      lastSnapshot = null
    }
    ```

    // Combat auto-save and load — subscribes to store changes and persists to SQLite.
    // D-15: persistence across app restarts. D-16: auto-save on every change.

    import { saveCombatState, loadCombatState, listCombats } from '@/shared/api'
    import type { CombatSnapshot } from '@/shared/api'
    import { useCombatantStore } from '@/entities/combatant'
    import { useConditionStore } from '@/entities/condition'
    import { useCombatTrackerStore } from '../model/store'
    import { hydrateManager, clearAllManagers, getManagerState } from './condition-bridge'
    import type { ConditionSlug } from '@engine'

    let unsubscribers: Array<() => void> = []
    let saveTimer: ReturnType<typeof setTimeout> | null = null

    function buildSnapshot(): CombatSnapshot | null {
      const tracker = useCombatTrackerStore.getState()
      if (!tracker.combatId || !tracker.isRunning) return null

      const combatants = useCombatantStore.getState().combatants
      const conditions = useConditionStore.getState().activeConditions

      return {
        id: tracker.combatId,
        name: 'Combat',
        round: tracker.round,
        turn: tracker.turn,
        activeCombatantId: tracker.activeCombatantId,
        isRunning: tracker.isRunning,
        combatants,
        conditions,
      }
    }

    function debouncedSave(): void {
      if (saveTimer) clearTimeout(saveTimer)
      saveTimer = setTimeout(async () => {
        const snapshot = buildSnapshot()
        if (snapshot) {
          try {
            await saveCombatState(snapshot)
          } catch (err) {
            console.error('Auto-save failed:', err)
          }
        }
      }, 300) // debounce 300ms to batch rapid changes
    }

    /** Subscribe to all combat-related store changes and auto-save. */
    export function setupAutoSave(): void {
      teardownAutoSave()

      unsubscribers.push(
        useCombatantStore.subscribe(debouncedSave),
        useConditionStore.subscribe(debouncedSave),
        useCombatTrackerStore.subscribe(debouncedSave)
      )
    }

    /** Unsubscribe from auto-save. */
    export function teardownAutoSave(): void {
      for (const unsub of unsubscribers) unsub()
      unsubscribers = []
      if (saveTimer) {
        clearTimeout(saveTimer)
        saveTimer = null
      }
    }

    /** Load the most recent running combat from SQLite and hydrate all stores. */
    export async function loadActiveCombat(): Promise<boolean> {
      try {
        const combats = await listCombats()
        const running = combats.find((c) => c.isRunning)
        if (!running) return false

        const snapshot = await loadCombatState(running.id)
        if (!snapshot) return false

        // Hydrate stores
        useCombatantStore.getState().setCombatants(snapshot.combatants)
        useCombatTrackerStore.getState().setCombatId(snapshot.id)
        useCombatTrackerStore.getState().setRound(snapshot.round)
        useCombatTrackerStore.getState().setTurn(snapshot.turn)
        useCombatTrackerStore.getState().setActiveCombatant(snapshot.activeCombatantId)

        if (snapshot.isRunning) {
          // Set isRunning via startCombat would reset round/turn, so set fields directly
          useCombatTrackerStore.setState({ isRunning: true })
        }

        // Hydrate ConditionManagers from persisted conditions
        clearAllManagers()
        const conditionsByComba = new Map<string, typeof snapshot.conditions>()
        for (const c of snapshot.conditions) {
          if (!conditionsByComba.has(c.combatantId)) {
            conditionsByComba.set(c.combatantId, [])
          }
          conditionsByComba.get(c.combatantId)!.push(c)
        }
        for (const [combatantId, conditions] of conditionsByComba) {
          hydrateManager(
            combatantId,
            conditions.map((c) => ({
              slug: c.slug as ConditionSlug,
              value: c.value ?? 1,
              isLocked: !!c.isLocked,
              grantedBy: c.grantedBy as ConditionSlug | undefined,
            }))
          )
        }

        return true
      } catch (err) {
        console.error('Failed to load combat:', err)
        return false
      }
    }
    ```

    import { useState, useCallback } from 'react'
    import { ChevronRight, ChevronLeft } from 'lucide-react'
    import { Button } from '@/shared/ui/button'
    import { useCombatTrackerStore } from '../model/store'
    import { advanceTurn, reverseTurn, canReverseTurn } from '../lib/turn-manager'
    import { useShallow } from 'zustand/react/shallow'

    export function TurnControls() {
      const { isRunning } = useCombatTrackerStore(
        useShallow((s) => ({ isRunning: s.isRunning }))
      )
      // Force re-render after turn change to update canReverse
      const [, setTick] = useState(0)

      const handleNext = useCallback(() => {
        advanceTurn()
        setTick((t) => t + 1)
      }, [])

      const handlePrevious = useCallback(() => {
        reverseTurn()
        setTick((t) => t + 1)
      }, [])

      if (!isRunning) return null

      return (
        <div className="flex items-center gap-1.5 p-2 border-t border-border/50">
          <Button
            variant="outline"
            size="sm"
            className="h-7 text-xs gap-1 flex-1"
            onClick={handlePrevious}
            disabled={!canReverseTurn()}
          >
            <ChevronLeft className="w-3.5 h-3.5" />
            Previous
          </Button>
          <Button
            size="sm"
            className="h-7 text-xs gap-1 flex-1"
            onClick={handleNext}
          >
            Next Turn
            <ChevronRight className="w-3.5 h-3.5" />
          </Button>
        </div>
      )
    }
