```typescript
export interface DiceEntry {
  sides: number   // e.g. 6, 20
  value: number   // rolled result, 1..sides
}

export interface Roll {
  id: string                  // crypto.randomUUID()
  formula: string             // original formula, e.g. "2d6+4"
  dice: DiceEntry[]           // each individual die rolled
  modifier: number            // sum of all flat +N/-N terms
  total: number               // sum of all dice values + modifier
  label?: string              // e.g. "Fireball damage"
  timestamp: number           // Date.now()
}

export function rollDice(formula: string, label?: string): Roll
```

```typescript
import { useEffect, useRef } from 'react'
import { toast } from 'sonner'
import { useRollStore } from '@/shared/model/roll-store'
import { RollResultToast } from './roll-result-toast'
```

Implementation approach (Claude's discretion per CONTEXT.md -- use `useEffect` with rolls length):
- `const rolls = useRollStore(state => state.rolls)` -- read the full rolls array.
- `const prevLengthRef = useRef(rolls.length)` -- track previous length to detect new rolls.
- `useEffect` watching `rolls.length`:
  - If `rolls.length > prevLengthRef.current` (new roll added):
    - Get the newest roll: `rolls[rolls.length - 1]`
    - Call `toast.custom((id) => <RollResultToast roll={newestRoll} />, { duration: 4000, position: 'bottom-right' })`
