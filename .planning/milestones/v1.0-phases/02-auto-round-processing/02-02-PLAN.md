    ```typescript
    function decrementDurationsForCreature(creatureId: string): void {
      const trackedDurations = creatureConditionDurations.value[creatureId]
      if (!trackedDurations) return

      const creature = creatures.value.find(c => c.id === creatureId)
      if (!creature) return

      const isProtected = protectedConditions.value[creatureId]?.length > 0

      // Loop through tracked durations and decrement
      for (const [conditionName, duration] of Object.entries(trackedDurations)) {
        // Skip protected conditions - they don't decrement per CONTEXT.md locked decision
        if (isProtected && protectedConditions.value[creatureId]?.includes(conditionName as Condition)) {
          continue
        }

        // Decrement duration
        const newDuration = duration - 1

        if (newDuration <= 0) {
          // Remove condition when duration hits 0
          const conditionIndex = creature.conditions.indexOf(conditionName as Condition)
          if (conditionIndex > -1) {
            creature.conditions.splice(conditionIndex, 1)
          }

          // Clear from tracked durations
          delete creature.conditionDurations?.[conditionName as Condition]
          delete creatureConditionDurations.value[creatureId]?.[conditionName]

          // Clean up empty creature entry
          if (Object.keys(creatureConditionDurations.value[creatureId] || {}).length === 0) {
            delete creatureConditionDurations.value[creatureId]
          }
        } else {
          // Update duration
          creature.conditionDurations?.(conditionName as Condition, newDuration)
          creatureConditionDurations.value[creatureId]![conditionName] = newDuration
        }
      }
    }
    ```
