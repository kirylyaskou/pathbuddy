1. Add a `reorderCreature` function that takes a source creature ID and a target creature ID, then swaps their initiative values so the sorted order changes:
   ```typescript
   function reorderCreature(sourceId: string, targetId: string): void {
     if (sourceId === targetId) return
     const source = creatures.value.find(c => c.id === sourceId)
     const target = creatures.value.find(c => c.id === targetId)
     if (!source || !target) return
     // Swap initiative values so sortedCreatures recomputes order
     const tempInit = source.initiative
     source.initiative = target.initiative
     target.initiative = tempInit
     // If initiatives are equal, also swap dexMod to ensure order change
     if (source.initiative === target.initiative) {
       const tempDex = source.dexMod
       source.dexMod = target.dexMod
       target.dexMod = tempDex
     }
   }
   ```

2. Export `reorderCreature` from the store return object.

**In src/components/CombatTracker.vue:**

1. Add a reactive ref to track the drag-over target for visual feedback:
   ```typescript

   ```

2. Add drag event handler functions:
   ```typescript
   const handleDragOver = (event: DragEvent) => {
     event.preventDefault()  // Required to allow drop
   }

   const handleDragEnter = (creatureId: string) => {
     dragOverCreatureId.value = creatureId
   }

   const handleDragLeave = () => {
     dragOverCreatureId.value = null
   }

   const handleDrop = (event: DragEvent, targetCreatureId: string) => {
     event.preventDefault()
     const sourceCreatureId = event.dataTransfer?.getData('text/plain')
     dragOverCreatureId.value = null
     if (sourceCreatureId && sourceCreatureId !== targetCreatureId) {
       combatStore.reorderCreature(sourceCreatureId, targetCreatureId)
     }
   }
   ```

3. Update each CreatureCard's wrapper in the template to be a drop target. Wrap each CreatureCard in a div that serves as the drop zone:
   ```html
   <div v-else class="space-y-2">
     <div
       v-for="(creature, index) in combatStore.sortedCreatures"
       :key="creature.id"
       @dragover.prevent="handleDragOver"
       @dragenter.prevent="handleDragEnter(creature.id)"
       @dragleave="handleDragLeave"
       @drop="handleDrop($event, creature.id)"
       :class="{ 'ring-2 ring-blue-400 ring-offset-2 rounded-lg': dragOverCreatureId === creature.id }"
     >
       <CreatureCard
         :creature="creature"
         :is-current-turn="creature.isCurrentTurn"
         :turn-index="index"
         :regeneration-disabled="getRegenerationDisabled(creature.id)"
         @modify-hp="handleModifyHp"

     </div>
   </div>
   ```
