```typescript
import type { ConditionSlug } from '../conditions/conditions'

export type ActionType = 'action' | 'reaction' | 'free' | 'passive'
export type ActionCost = 1 | 2 | 3 | null
export type ActionCategory = 'offensive' | 'defensive' | 'interaction' | null
export type DegreeKey = 'critical_success' | 'success' | 'failure' | 'critical_failure'

export interface ActionOutcome {
  conditions?: Array<{ slug: ConditionSlug; value?: number }>
  damage?: string
  effect?: string
}

export type ActionOutcomeMap = Partial<Record<DegreeKey, ActionOutcome>>

export interface Action {
  slug: string
  name: string
  actionType: ActionType
  cost: ActionCost
  category: ActionCategory
  traits: string[]
  outcomes?: ActionOutcomeMap
}
```

**Part B: Create engine/actions/action-data.ts**

Read ALL 545 JSON files from `refs/pf2e/actions/` recursively (including nested subdirectories under class/, archetype/, ancestry/, etc. per Pitfall 6). For each JSON file, extract:
- `slug`: derive from filename (strip .json, e.g., "strike.json" -> "strike")
- `name`: from JSON `name` field
- `actionType`: from `system.actionType.value`
- `cost`: from `system.actions.value` (null for reactions/free/passive)
- `category`: from `system.category` (null if absent)
- `traits`: from `system.traits.value` array

Write a script (inline in the task, not a separate file) that:
1. Recursively reads all .json files under `refs/pf2e/actions/`
2. For each file, parses the JSON and extracts the 6 fields above
3. Generates a TypeScript source file `engine/actions/action-data.ts`

The generated file should contain:

```typescript
import type { Action } from './types'
import { ACTION_OUTCOMES } from './action-outcomes'

// ─── Action Data (D-01) ─────────────────────────────────────────────────────
// Auto-generated from refs/pf2e/actions/ (545 entries).
// Combat-relevant outcomes merged from ACTION_OUTCOMES.
// DO NOT EDIT BY HAND — regenerate from refs/ data.

const RAW_ACTIONS: Omit<Action, 'outcomes'>[] = [
  { slug: 'strike', name: 'Strike', actionType: 'action', cost: 1, category: 'offensive', traits: ['attack'] },
  { slug: 'demoralize', name: 'Demoralize', actionType: 'action', cost: 1, category: 'offensive', traits: ['auditory', 'concentrate', 'emotion', 'fear', 'mental'] },
  // ... all 545 entries
]

/** Complete action lookup map. Entries with combat outcomes have `outcomes` attached from ACTION_OUTCOMES. */
export const ACTIONS: ReadonlyMap<string, Action> = new Map(
  RAW_ACTIONS.map(raw => [
    raw.slug,
    { ...raw, ...(ACTION_OUTCOMES[raw.slug] ? { outcomes: ACTION_OUTCOMES[raw.slug] } : {}) },
  ])
)
```

The script approach: Write a Node.js script to stdout or directly to the file. Use `fs.readdirSync` recursively on `refs/pf2e/actions/`. For each `.json` file:
- Read and parse
- Extract slug from filename (path.basename without .json)
- Extract name, actionType, cost, category, traits
- Add to sorted array

Write the TypeScript output to `engine/actions/action-data.ts`.
